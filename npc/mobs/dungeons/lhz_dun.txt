		donpcevent "summon_boss_lt::Onsummon";
		stopnpctimer;

	end;

	monster "lhz_dun03",.@x2,.@y2,strmobinfo(1,.@mob2),.@mob2,1,"summon_boss_lt::OnMVP";
	end;

OnMyMvPDead:
