-	shop	Goldpc_Points	-1,512:-1 //生成商店，腳本名字需要和下面腳本中名字叫對應
//腳本開始
//prontera,166,178,4	script	線上積分商店	564,{


-	script	 GoldpcIcon	-1,{

OnPCGoldTimerOpenFilter:
	unlockcmd;
	processhalt;
	message strcharinfo(0),"此商店只能使用[線上積分]進行交易."; //此處告訴玩家此商店使用什麼作為商店貨幣
	dispbottom "您現在還有線上積分 [ "+Goldpc_Points+" ] 點."; //有多少點數
	callshop "Goldpc_Points",1; //開啟商店
	npcshopattach "Goldpc_Points"; //關聯腳本
	end;
//購買事件
OnBuyItem:
	set .@i,0; //初始化購買個數為0
	while (.@i < getarraysize(@bought_nameid)) {
		set .@j, 0;
		while (.@j < getarraysize(.Commodity)) {
			if(.Commodity[.@j] == @bought_nameid[.@i]) {
				set @itemcost, .Price[.@j] * @bought_quantity[.@i];
				set @totalcost, @totalcost + @itemcost;
				break; 
			}
			set .@j, .@j+1; 
		}
		set .@i, .@i+1; 
	}
	if (@totalcost > Goldpc_Points) dispbottom "你沒用足夠的[ 線上積分 ]點.";
	else {
		set .@i,0;
		while (.@i < getarraysize(@bought_nameid)) {
			getitem @bought_nameid[.@i], @bought_quantity[.@i];
			dispbottom "成功購買[ "+@bought_quantity[.@i]+" ]個[ "+getitemname(@bought_nameid[.@i])+" ]";
			set .@i, .@i+1; 
		}
		//購買成功，扣除點數
		set Goldpc_Points,Goldpc_Points - @totalcost; 
		dispbottom "您消費了[ "+@totalcost+" ]點.還剩[ "+Goldpc_Points+" ]點";
	}
	//初始化玩家變量
	set @totalcost, 0;
	deletearray @bought_nameid[0],128;
	deletearray @bought_quantity[0],128;
	end;

OnInit:
//---------------------------------------------------
	waitingroom "線上積分商店",0;
	//設置商品的ID
	setarray .Commodity[0],
	12822,12411,12710,14539,14540,23374,23375,23376,23377,12220,	//卷軸
	12622,7776,25793,12922,12912,12902,61057,	//輔助道具
	1000319,1000345,1000875,1001038,1000365,1001069,1001175,1001252;	//服飾特效石


	//設置商品價格[對應商品ID]
	setarray .Price[0],
	2,2,2,2,2,2,2,2,2,2,	//卷軸
	30,30,30,10,15,20,30,	//輔助道具
	120,120,120,120,120,120,120,120;	//服飾特效石

// --------------------------------------------------

//初始化商店中物品
	debugmes "線上積分商城加載中...";
	set .@j,0;
	if (getarraysize(.Commodity) != getarraysize(.Price))
	{
		debugmes "[錯誤],您設置的商品ID與商品價格不對應";
		debugmes "[錯誤],商品ID共["+getarraysize(.Commodity)+"],商品價格["+getarraysize(.Price)+"]";
		end;
	}
	while (.@j < getarraysize(.Commodity)) {
		npcshopdelitem "Goldpc_Points",512;
		npcshopadditem "Goldpc_Points",.Commodity[.@j],.Price[.@j];
		set .@j,.@j+1; 
	}
	debugmes "【線上積分商城】商品"+getarraysize(.Commodity)+"個";
	end;
}
