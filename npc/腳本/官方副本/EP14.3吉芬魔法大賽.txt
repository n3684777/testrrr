			mes "還是等到 Lv.140 後再來吧…";

			close;

		mes ""+.@md_name$+"已經創立了∼";
		mes "請選擇^0066CC進入"+.@md_name$+"^000000。";

			mes "裡面是很危險的…"; 
			mes "還是等到 Lv.140 後再來吧…";
